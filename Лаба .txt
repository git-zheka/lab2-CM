double Nioton(double a, double b, double eps, double i2, double c) {

    do {
     c = a - ((b - a) / (f(b) - f(a)) * f(a));
     b = c;
     i2++;
    } while (abs(f(c)) > eps);

     cout << "Метод Ньотона X = " << c << "\nЦиклів - " << i2<< endl;
    
     return 0;
}

double f(double x) {
    return pow(x, 3) + 3 * pow(x, 2) + 12 * x - 3;
}

double g(double x) {
  return  pow(3 - 12*x - 3 * pow(x, 2), 1/3);
}

double f1d(double x) {
    return 3 * pow(x, 2) + 6 * x + 12;
}

double f2d(double x) {
    return 6 * x + 6;
